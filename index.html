Mudar a cor da aba do navegador para a cor azul

<head>
    <!-- Cor da aba para navegadores móveis -->
    <meta name="theme-color" content="#1e90ff"> <!-- Azul claro -->

    <!-- Para navegadores no Android -->
    <meta name="msapplication-navbutton-color" content="#1e90ff">

    <!-- Para navegadores no iOS -->
    <meta name="apple-mobile-web-app-status-bar-style" content="black-translucent">
</head>
